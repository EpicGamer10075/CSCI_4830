#

validates that the break within the timer form is within the main timer’s duration

function validateForm() { 
	let timerForm = document.forms["timer"];
	let tdur = Number(timerForm["tdur"].value);
	if(timerForm["bstr"] != null) {
		let bstr = Number(timerForm["bstr"].value);
		let bdur = Number(timerForm["bdur"].value);
		if(tdur <= (bstr + bdur)) { 
			alert("Break must be contained within timer");
			return false;
		}
	}
}

let breakID = 1;
#

creates new break within timer form

function newBreak(){ 
	let focusForm = document.querySelector(".forms-card"); 
	let curForm = focusForm.innerHTML; 
	curForm = curForm.substring(0, curForm.lastIndexOf("required")+20);
	focusForm.innerHTML = curForm +`<div class="form-group"><label for="bstr">Break start:</label>
									<input type="number" id="bstr" name="bstr" value="60" min="0" max="100" required></div>
									<div class="form-group"><label for="bdur">Break duration:</label>
									<input type="number" id="bdur" name="bdur" value="60" min="0" max="100" required></div>
									<button type="button" onclick="delBreak()" class="btn-secondary" style="margin-bottom: 12px; width: 100%;">Delete Break</button>
									<input type="submit" value="Submit" class="btn" style="width: 100%;">`;
	breakID++;
}
#

removes last break within timer form

function delBreak(){ 
	let focusForm = document.querySelector(".forms-card"); 
	let curForm = focusForm.innerHTML; 
	curForm = curForm.substring(0, curForm.lastIndexOf("<div class=\"form-group\"><label for=\"bstr"));
	focusForm.innerHTML = curForm +`<button type="button" onclick="newBreak()" class="btn-secondary" style="margin-bottom: 12px; width: 100%;">Add Break</button>
									<input type="submit" value="Submit" class="btn" style="width: 100%;">`; 
	breakID--;
}
